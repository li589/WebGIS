### 1.2 后端 Bug 详情

[export_overlay_assets.py L373-374](file:///d:/temp_desktop/Proj/Comprehensive%20Geographic%20Data%20Analysis%20system/Tools/export_overlay_assets.py#L373-L374)：

```python
lat = m.get("lat") or m.get("latitude")     # ❌ numpy array 不能用 Python `or`
lon = m.get("lon") or m.get("longitude")    # ❌ 同上
```

崩溃日志：
```
ValueError: The truth value of an array with more than one element is ambiguous.
Use a.any() or a.all()
```

**根因**：`m.get("lat")` 返回 numpy array（多元素），Python 的 `or` 运算符尝试对 array 求布尔值，引发 `ValueError`。

const MIN_PIXEL_FOR_LABEL = 60

function shouldShowLabel(p1Screen: {x:number,y:number}, p2Screen: {x:number,y:number}): boolean {
  const dx = p2Screen.x - p1Screen.x
  const dy = p2Screen.y - p1Screen.y
  const pixelDist = Math.sqrt(dx * dx + dy * dy)
  return pixelDist >= MIN_PIXEL_FOR_LABEL
}

**事件处理**：

```typescript
function onClick(e: MapMouseEvent) {
  if (options.getInteractionMode() !== 'measure') return
  options.addMeasurePoint({ lng: e.lngLat.lng, lat: e.lngLat.lat })
  updateGeoJSON()
  canvas.updateState(...)
}

function onDblClick(e: MapMouseEvent) {
  if (options.getInteractionMode() !== 'measure') return
  e.preventDefault()  // 阻止默认双击缩放
  options.completeMeasure()
  updateGeoJSON()
}

function onContextMenu(e: MapMouseEvent) {
  if (options.getInteractionMode() !== 'measure') return
  e.preventDefault()  // 阻止右键菜单
  options.undoLastMeasurePoint()
  updateGeoJSON()
}

function onMouseMove(e: MapMouseEvent) {
  if (options.getInteractionMode() !== 'measure') return
  if (!options.getMeasureState().isDrawing) return
  options.setHoverPoint({ lng: e.lngLat.lng, lat: e.lngLat.lat })
  updateGeoJSON()
  canvas.requestRedraw()
}
```

**模式切换处理**：

```typescript
function applyMeasureMode() {
  const isMeasure = options.getInteractionMode() === 'measure'
  if (isMeasure) {
    options.map.dragPan.disable()
    options.map.doubleClickZoom.disable()
    options.map.boxZoom.disable()
    canvas.show()
  } else {
    options.map.doubleClickZoom.enable()
    options.map.boxZoom.enable()
    // 注意：dragPan 由 mapInteractionModule.applyInteractionMode() 管理（select 也禁用）
    canvas.hide()
  }
}
```
