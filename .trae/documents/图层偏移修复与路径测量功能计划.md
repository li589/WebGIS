```python
def _render_png(data, png_path, cmap="viridis", vmin=None, vmax=None, transparent=True):
    png_path.parent.mkdir(parents=True, exist_ok=True)
    data = np.asarray(data, dtype=np.float64)
    if data.ndim != 2:
        raise ValueError(f"Expected 2D array, got {data.ndim}D")
    n_lat, n_lon = data.shape
    # 关键修复：figsize 与数据行列严格 1:1（dpi=100 → 1 inch = 100 px）
    # 最小 1.0 英寸避免极小数组渲染失败，但不再钳位到 2 英寸
    figsize = (max(n_lon / 100, 1.0), max(n_lat / 100, 1.0))
    fig = plt.figure(figsize=figsize, frameon=False)
    ax = fig.add_axes([0, 0, 1, 1])
    ax.set_axis_off()
    masked = np.ma.masked_invalid(data)
    ax.imshow(masked, cmap=cmap, vmin=vmin, vmax=vmax,
              aspect="auto", origin="upper", interpolation="nearest")
    fig.savefig(png_path, dpi=100, transparent=transparent, pad_inches=0)
    plt.close(fig)
    print(f"  [OK] PNG saved: {png_path.name} ({n_lat}x{n_lon})")
```

#### 修改 2：新增 `_bounds_from_centers()` 工具函数

**文件**：[Tools/export_overlay_assets.py](file:///d:/temp_desktop/Proj/Comprehensive%20Geographic%20Data%20Analysis%20system/Tools/export_overlay_assets.py)（在 `_write_bounds` 后新增）

**用途**：将像素中心坐标数组转换为像素边界 bounds，处理半像素扩展：

```python
def _bounds_from_centers(lat_1d, lon_1d):
    """从像素中心坐标计算地理边界 (west, south, east, north)。
    
    lat_1d: 1D 纬度数组（升序或降序均可）
    lon_1d: 1D 经度数组（升序或降序均可）
    """
    lat = np.asarray(lat_1d).ravel()
    lon = np.asarray(lon_1d).ravel()
    # 计算像素间距（取相邻差的中位数，避免边界跳变）
    dlat = float(np.median(np.abs(np.diff(lat)))) if len(lat) > 1 else 0.0
    dlon = float(np.median(np.abs(np.diff(lon)))) if len(lon) > 1 else 0.0
    north = float(max(lat[0], lat[-1]) + dlat / 2)
    south = float(min(lat[0], lat[-1]) - dlat / 2)
    east = float(max(lon[0], lon[-1]) + dlon / 2)
    west = float(min(lon[0], lon[-1]) - dlon / 2)
    return (west, south, east, north)
```

**新增** `_reproject_ease_to_wgs84()`：

```python
# EASE-Grid 2.0 9km 标准定义（NSIDC）
_EASE_GRID_9K_CRS = "EPSG:6933"

def _ease_grid_9k_transform():
    """返回 EASE-Grid 2.0 9km 的 Affine 变换。
    
    标准参数（NSIDC）：
    - 上左角 x: -17367530.45 m
    - 上左角 y: 7314540.83 m
    - 像素大小: 9000.879 m（行列相同）
    """
    from rasterio.transform import from_origin
    return from_origin(-17367530.45, 7314540.83, 9000.879, 9000.879)

def _reproject_ease_to_wgs84(data, target_resolution=0.1):
    """将 EASE-Grid 2.0 9km 数据重投影到 WGS84 等经纬度网格。
    
    Args:
        data: (n_lat, n_lon) 2D array，EASE-Grid 2.0 9km
        target_resolution: 目标分辨率（度），默认 0.1°
    
    Returns:
        (reprojected_data, bounds) — bounds = (west, south, east, north)
    """
    from rasterio.warp import reproject, calculate_default_transform
    from rasterio.enums import Resampling
    
    src_transform = _ease_grid_9k_transform()
    src_crs = _EASE_GRID_9K_CRS
    dst_crs = "EPSG:4326"
    
    n_lat, n_lon = data.shape
    # 计算目标网格
    dst_transform, dst_width, dst_height = calculate_default_transform(
        src_crs, dst_crs, n_lon, n_lat, *src_transform * (0, 0, n_lon, n_lat),
        resolution=target_resolution
    )
    
    dst_data = np.full((dst_height, dst_width), np.nan, dtype=np.float64)
    reproject(
        source=data,
        destination=dst_data,
        src_transform=src_transform,
        src_crs=src_crs,
        dst_transform=dst_transform,
        dst_crs=dst_crs,
        resampling=Resampling.nearest,  # 分类数据用最近邻；连续数据可改 bilinear
        src_nodata=np.nan,
        dst_nodata=np.nan,
    )
    
    # 计算 WGS84 边界
    west = dst_transform[2]
    east = dst_transform[2] + dst_transform[0] * dst_width
    north = dst_transform[5]
    south = dst_transform[5] + dst_transform[4] * dst_height
    
    return dst_data, (float(west), float(south), float(east), float(north))
```

#### 步骤 2：新增 `measure-geo.ts`（测地线计算）

**文件**：`Code/frontend/src/components/map/measure-geo.ts`（新增）

```typescript
/** Haversine 测地线距离（米） */
export function haversineDistance(p1: {lng:number,lat:number}, p2: {lng:number,lat:number}): number {
  const R = 6371008.8  // 地球平均半径（米）
  const toRad = (d: number) => d * Math.PI / 180
  const dLat = toRad(p2.lat - p1.lat)
  const dLon = toRad(p2.lng - p1.lng)
  const a = Math.sin(dLat/2)**2 + Math.cos(toRad(p1.lat)) * Math.cos(toRad(p2.lat)) * Math.sin(dLon/2)**2
  return 2 * R * Math.asin(Math.sqrt(a))
}

/** 初始方位角（度，0=北，顺时针） */
export function bearing(p1: {lng:number,lat:number}, p2: {lng:number,lat:number}): number {
  const toRad = (d: number) => d * Math.PI / 180
  const toDeg = (r: number) => r * 180 / Math.PI
  const φ1 = toRad(p1.lat), φ2 = toRad(p2.lat), Δλ = toRad(p2.lng - p1.lng)
  const y = Math.sin(Δλ) * Math.cos(φ2)
  const x = Math.cos(φ1)*Math.sin(φ2) - Math.sin(φ1)*Math.cos(φ2)*Math.cos(Δλ)
  return (toDeg(Math.atan2(y, x)) + 360) % 360
}

/** 格式化距离：m → "1.2 km" 或 "850 m" */
export function formatDistance(m: number): string {
  if (m < 1000) return `${m.toFixed(0)} m`
  if (m < 10000) return `${(m/1000).toFixed(2)} km`
  return `${(m/1000).toFixed(1)} km`
}

/** 格式化角度：度 → "45.3°" */
export function formatBearing(deg: number): string {
  return `${deg.toFixed(1)}°`
}

/** 计算路径总距离与各段信息 */
export interface SegmentInfo {
  distance: number
  bearing: number
  midpoint: { lng: number; lat: number }
}
export function computeSegments(points: Array<{lng:number,lat:number}>): { segments: SegmentInfo[]; total: number } {
  const segments: SegmentInfo[] = []
  let total = 0
  for (let i = 0; i < points.length - 1; i++) {
    const dist = haversineDistance(points[i], points[i+1])
    const bear = bearing(points[i], points[i+1])
    const mid = { lng: (points[i].lng + points[i+1].lng)/2, lat: (points[i].lat + points[i+1].lat)/2 }
    segments.push({ distance: dist, bearing: bear, midpoint: mid })
    total += dist
  }
  return { segments, total }
}
```

// 像素距离阈值：当段在屏幕上的像素长度 < 阈值时隐藏标签
const MIN_PIXEL_FOR_LABEL = 60  // 60px 以下不显示标签

function shouldShowLabel(p1: ScreenPoint, p2: ScreenPoint, zoom: number): boolean {
  const dx = p2.x - p1.x, dy = p2.y - p1.y
  const pixelDist = Math.sqrt(dx*dx + dy*dy)
  // 高缩放级别（放大）时阈值降低，低缩放级别（缩小）时阈值升高
  const threshold = MIN_PIXEL_FOR_LABEL * Math.max(0.5, 1 - (zoom - 3) * 0.05)
  return pixelDist >= threshold
}

      options.map.doubleClickZoom.enable()
      canvas?.hide()

## 四、假设与决策

### 4.1 问题 1 决策

| 决策 | 选择 | 理由 |
|------|------|------|
| EASE-Grid 处理方式 | 重投影到 WGS84 再渲染 | 用户明确选择；位置完全准确；插值代价可接受 |
| 重投影分辨率 | 0.1° | 平衡精度与文件大小；全球 0.1° 约 1800×3600，PNG ~6.5MB |
| 重投影重采样方法 | `Resampling.nearest` | 保留原始像素值，避免分类数据（如 landcover）产生无效插值；对连续数据（omega/biomass）影响轻微 |
| EASE-Grid 像素大小 | 9000.879m（NSIDC 标准） | Resolution_meters=9000 是近似值；Forest_Ratio.mat 的 Transform 像素=5004 疑似保存错误，以 NSIDC 标准为准 |
| 是否修改前端 | **否** | [overlay-image-module.ts](file:///d:/temp_desktop/Proj/Comprehensive%20Geographic%20Data%20Analysis%20system/Code/frontend/src/components/map/overlay-image-module.ts) 的 `coordinates` 映射正确，只需后端 bounds 准确即可 |
| GPCP 0.5° 全球层 bounds | 保留 (-180, -90, 180, 90) | GPCP 像素中心 -179.75~179.75，半像素扩展后正好 -180~180，当前 bounds 正确 |

### 4.2 问题 2 决策

| 决策 | 选择 | 理由 |
|------|------|------|
| 模式方案 | 新增 `measure` 模式 | 用户确认；保留 `select` 点气象查询 |
| 测地线计算 | 自实现 Haversine | 避免引入 turf.js 依赖；函数简单（<30 行） |
| 路径渲染 | MapLibre GeoJSON Source + Layer | 线条/圆点原生支持，跟随地图缩放 |

| 双击缩放 | measure 模式下禁用 | 避免与双击完成路径冲突 |
| 右键菜单 | measure 模式下禁用 | 避免与右键撤销冲突 |
