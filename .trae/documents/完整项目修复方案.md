### 1.3 weather_providers_repository.py 加密静默降级（安全漏洞）

**位置**：`Code/backend/app/services/weather_providers_repository.py:79-97`（`_encrypt`）

**修复**：抽取通用加密工具，三个仓库统一复用。

#### 步骤 1：新建 `Code/backend/app/services/_secrets_crypto.py`

```python
"""统一的密钥加解密工具，所有 repository 共用。

设计原则：
- 生产环境（environment == "production"）强制加密，失败即 raise
- 开发环境允许明文降级（仅当 encryption_key 为空时）
- test 环境完全跳过加密
"""
from __future__ import annotations

import base64
import logging
import os
from typing import Literal

logger = logging.getLogger(__name__)

EnvLevel = Literal["production", "staging", "development", "test"]


def secrets_encryption_required(environment: EnvLevel) -> bool:
    """生产环境必须加密，其他环境可选。"""
    return environment == "production"


def encrypt_secret(plaintext: str, encryption_key: str, environment: EnvLevel) -> tuple[str, str]:
    """加密密钥，返回 (ciphertext_b64, iv_b64)。
    
    失败行为：
    - production 环境：抛 RuntimeError（fail-fast）
    - 其他环境：log warning 后返回明文（ciphertext=plaintext, iv=""）
    """
    if not encryption_key:
        if secrets_encryption_required(environment):
            raise RuntimeError(
                "Encryption key is required in production environment. "
                "Set BACKEND_GEE_CREDENTIALS_ENCRYPTION_KEY to a 32-byte hex string."
            )
        logger.warning("Encryption key not set, storing secret as plaintext (env=%s)", environment)
        return plaintext, ""
    
    try:
        from cryptography.hazmat.primitives.ciphers.aead import AESGCM
        key_bytes = bytes.fromhex(encryption_key)
        iv = os.urandom(12)
        aesgcm = AESGCM(key_bytes)
        ct = aesgcm.encrypt(iv, plaintext.encode("utf-8"), None)
        return base64.b64encode(ct).decode("ascii"), base64.b64encode(iv).decode("ascii")
    except ImportError:
        msg = "cryptography package not installed"
        if secrets_encryption_required(environment):
            raise RuntimeError(msg) from None
        logger.warning("%s, storing secret as plaintext", msg)
        return plaintext, ""
    except Exception as e:
        if secrets_encryption_required(environment):
            raise RuntimeError(f"Encryption failed: {e}") from e
        logger.error("Encryption failed, storing as plaintext: %s", e)
        return plaintext, ""


def decrypt_secret(ciphertext_b64: str, iv_b64: str, encryption_key: str) -> str:
    """解密密钥。"""
    if not encryption_key or not iv_b64:
        return ciphertext_b64  # 明文存储的旧数据
    try:
        from cryptography.hazmat.primitives.ciphers.aead import AESGCM
        key_bytes = bytes.fromhex(encryption_key)
        iv = base64.b64decode(iv_b64)
        ct = base64.b64decode(ciphertext_b64)
        return AESGCM(key_bytes).decrypt(iv, ct, None).decode("utf-8")
    except Exception as e:
        logger.error("Decryption failed: %s", e)
        raise
```

def _safe_sqrt(x: float) -> float:
    """非负开方，负值或 NaN 返回 0.0（物理上介电常数的虚部/实部应非负）。"""
    if not math.isfinite(x) or x < 0:
        return 0.0
    return math.sqrt(x)

# L357-360 改为：

znb = _safe_sqrt((mag + epwbx) / 2)

### 2.1 ndvi.py 逐像素循环向量化（性能 100×）

**位置**：`Code/algorithms/providers/Python/algorithms/ndvi.py:133-142`

**上下文**：数据是三维 `(rows, cols, time)`，L108-147 `process_ndvi_stack_to_daily` 函数。

**修复**：利用 `savgol_filter` 支持沿 axis 处理 N 维数组的特性。

```python
def process_ndvi_stack_to_daily(/* 原参数 */):
    # ... 原数据准备逻辑 ...
    
    # 旧代码（L133-142）：
    # for pixel_index in range(flattened.shape[0]):
    #     daily_flattened[pixel_index] = vi_sg_interpolate(...)
    
    # 新代码：向量化处理
    # 1. 统计每个像素的有效点数
    valid_counts = valid_mask.sum(axis=1)  # shape: (n_pixels,)
    
    # 2. 对有效点充足的像素批量做 SG 滤波
    sufficient_mask = valid_counts >= sg_window_length  # shape: (n_pixels,)
    if np.any(sufficient_mask):
        # 提取充足像素的数据（用 0 填充 NaN 以满足 savgol_filter 输入要求）
        data_sufficient = flattened[sufficient_mask].copy()
        nan_mask = np.isnan(data_sufficient)
        data_sufficient[nan_mask] = 0.0  # 临时填充
        
        # 沿时间轴（axis=1）批量滤波
        filtered = savgol_filter(
            data_sufficient, sg_window_length, sg_polyorder, axis=1, mode="interp"
        )
        
        # 恢复 NaN 位置（滤波后的 NaN 区域仍设为 NaN）
        filtered[nan_mask] = np.nan
        
        # 线性插值到 output_days
        # 这里仍需逐像素插值（因为每个像素的 sg_days 可能不同），
        # 但滤波这一步已向量化，性能提升主要来自这里
        for idx, pixel_idx in enumerate(np.where(sufficient_mask)[0]):
            daily_flattened[pixel_idx] = _linear_interp_with_nan(
                sg_days, filtered[idx], output_days
            )
    
    # 3. 对有效点不足的像素降级为线性插值或 NaN
    insufficient_mask = (~sufficient_mask) & (valid_counts >= 2)
    for pixel_idx in np.where(insufficient_mask)[0]:
        valid_indices = valid_mask[pixel_idx]
        daily_flattened[pixel_idx] = _linear_interp_with_nan(
            sg_days[valid_indices], flattened[pixel_idx][valid_indices], output_days
        )
    # valid_counts < 2 的像素保持 NaN（已初始化）
```

**验证**：
1. 构造 100×100×46 的测试数据，对比向量化前后耗时（预期 100× 提升）

新建 `Code/backend/app/services/_sqlite_pool.py`：

```python
"""SQLite 线程局部连接池。

SQLite 连接不能跨线程共享，但同一线程内可复用。
使用 threading.local() 为每个线程缓存一个连接。
"""
from __future__ import annotations

import sqlite3
import threading
from pathlib import Path
from typing import Any


class ThreadLocalSQLitePool:
    def __init__(self, db_path: str | Path, init_schema: Any | None = None):
        self.db_path = Path(db_path)
        self._init_schema = init_schema  # callable(conn) -> None
        self._local = threading.local()
        self._lock = threading.Lock()
        self._initialized = False
    
    def _connect(self) -> sqlite3.Connection:
        conn = sqlite3.connect(str(self.db_path), timeout=30.0, check_same_thread=False)
        conn.row_factory = sqlite3.Row
        # 启用 WAL 模式（写入不阻塞读取）
        conn.execute("PRAGMA journal_mode=WAL")
        conn.execute("PRAGMA synchronous=NORMAL")  # 平衡性能与安全
        conn.execute("PRAGMA foreign_keys=ON")
        return conn
    
    def get_connection(self) -> sqlite3.Connection:
        """获取当前线程的连接，首次调用时创建并初始化 schema。"""
        if not hasattr(self._local, "conn"):
            with self._lock:
                conn = self._connect()
                if not self._initialized and self._init_schema:
                    self._init_schema(conn)
                    self._initialized = True
                self._local.conn = conn
        return self._local.conn
    
    def close_thread_connection(self) -> None:
        """关闭当前线程的连接（用于线程退出或测试清理）。"""
        if hasattr(self._local, "conn"):
            self._local.conn.close()
            del self._local.conn
```

修改各 repository：

```python

class ApiKeysRepository:
    def __init__(self, db_path, encryption_key="", environment="development"):
        self._pool = ThreadLocalSQLitePool(db_path, init_schema=self._init_schema)
        # ...
    
    def _init_schema(self, conn):
        conn.execute("""CREATE TABLE IF NOT EXISTS ...""")
        conn.commit()
    
    # 原 _connect 方法删除，改为：
    def _connect(self):
        return self._pool.get_connection()
    
    # 注意：with self._connect() as conn 会调用 conn.__enter__/__exit__，
    # sqlite3.Connection 的上下文管理器会 commit/rollback 但不关闭连接，

```

export interface FetchOptions extends RequestInit {

export async function requestJson<T = unknown>(
  path: string,
  options: FetchOptions = {},
): Promise<T> {
  const {
    parseJson = true,
    prefixBaseUrl = true,
    headers: customHeaders = {},
    ...rest
  } = options;

    throw new Error(`HTTP ${response.status}: ${message}`);
  }
  
  if (!parseJson) {

  }
  
  return response.json() as Promise<T>;
}

**位置**：`cache_store.py`（63 行）

**修复**：
```python
import os
import shutil
from cachetools import LRUCache
from pathlib import Path

class CacheStore:
    def __init__(self, cache_dir: Path, max_size_bytes: int = 10 * 1024**3):  # 默认 10GB
        self.cache_dir = cache_dir
        self.max_size_bytes = max_size_bytes
        self._index = LRUCache(maxsize=10000)  # 内存索引
        self._current_size = 0
        self._load_index()
    
    def put(self, key: str, source: Path) -> None:
        target = self.cache_dir / key
        # 硬链接（同文件系统）或复制
        try:
            os.link(source, target)
        except OSError:
            shutil.copyfile(source, target)  # 不复制元数据，更快
        
        self._current_size += target.stat().st_size
        self._index[key] = target
        self._evict_if_needed()
    
    def _evict_if_needed(self):
        while self._current_size > self.max_size_bytes and self._index:
            key, path = self._index.popitem(last=False)  # LRU
            try:
                self._current_size -= path.stat().st_size
                path.unlink()
            except FileNotFoundError:
                pass
```
