### 5.2 创建统一 tile 服务

**新建文件**: `Code/backend/app/services/unified_tile_service.py`

```python
from typing import Protocol

class TileProvider(Protocol):
    """瓦片提供者接口。"""
    def get_tile(self, layer_id: str, z: int, x: int, y: int, **params) -> "TileResponse": ...

@dataclass
class TileResponse:
    data: bytes
    content_type: str
    cache_ttl: int = 3600

class UnifiedTileService:
    """统一瓦片服务：按 layer_id 路由到具体 tile provider。
    
    Provider 注册：
    - WeatherTileProvider: 调用现有 tile_service.py 生成 GeoJSON
    - BasemapTileProvider: 调用现有 tile_proxy_service.py 代理 raster
    """
    
    def __init__(self):
        self._providers: dict[str, TileProvider] = {}
    
    def register_provider(self, layer_id_pattern: str, provider: TileProvider) -> None: ...
    
    def get_tile(self, layer_id: str, z: int, x: int, y: int, **params) -> TileResponse:
        provider = self._resolve_provider(layer_id)
        if provider is None:
            raise HTTPException(404, f"No tile provider for layer: {layer_id}")
        return provider.get_tile(layer_id, z, x, y, **params)
```
